【注册登录模块】

1 注册

#链接#  http://43.254.55.158/shipService/api/v1/user/register  【POST】
#参数#  param={phone:123475,password:132245,code:123456,username:12,shipname:32,shipno=87} 
       （手机和密码,验证码不能为空，其他选填）
#返回值# 
1. 失败 {"code":"400","msg":"请求参数错误"}
2. 成功 {"code":"200","msg":"用户注册成功"}

2 登录

#链接#  http://43.254.55.158/shipService/api/v1/user/login     【POST】
#参数#  phone=13444444&password=132245&deviceToken=yuuhjioj
#返回值# 
1. 失败 {"code":"400","msg":"请求参数错误"}
2. 成功 {"code":"200","msg":"用户登陆成功","result":{"balance":0.00,"city":"上海","couponCount":0,"id":21,"orderCount":0,"phone":"18516293301","registerDate":1441462403000,"status":0,"token":"c4929be9915d2eaf4cb5bd273fa38ab9","url":"43.254.55.158/shipService/headImage/1441370126804.jpg"}}

3 获取验证码

#链接#  http://43.254.55.158/shipService/api/v1/user/code
#参数#  phone=18516293301     一分钟失效时间
#返回值# 
1. 失败 {"code":"400","msg":"请求参数错误"}
2. 成功 {"code":"200","msg":"验证码","result":"655322"}

4 重置密码

#链接#  http://43.254.55.158/shipService/api/v1/user/update
#参数#  phone=18516293301&password=111111&code=123456
#返回值# 
1. 失败 {"code":"400","msg":"验证码过期"}
2. 成功 {"code":"200","msg":"重置成功"}

5 登出

#链接#  http://43.254.55.158/shipService/api/v1/user/logout
#参数#  id=1&token=8bdbda6aa3344c3a20808d77ad85ddda
#返回值# 
1. 失败 {"code":"400","msg":"请求参数错误"}
2. 成功 {"code":"200","msg":"登出成功"}

#参数#  
	userId		用户ID（必填）
#返回值#

3 银行卡绑定预校验(POST)


#链接#  http://43.254.55.158/shipService/api/v1/bank/pfs
#参数#  
	userId				
	bankCode			机构代码（必填）
	bankName			机构名称（必填）
	bankCardType			银行卡类型（必填）
	bankCardNo			银行卡号明文（必填）
	realName			用户真实姓名，应与银行卡账户姓名保持一致（必填）
	idNo				证件号码, 与银行预留一致（必填）
	idType				证件类型，（必填）
	mobileNo			银行卡预留手机号（必填）
	cvv2				信用卡CVV2，信用卡必传，规则：三位数字（选填）
	validThru			信用卡有效期: yy/mm，信用卡必传（选填）


#返回值# 
    requestNo			签约请求号
示例  {"code":"200","msg":"请求完成","result":{"requestNo":"eebf8d24-1ef9-4e68-bf80-1d8ade-123"}}
